49class BREZEL_API SigmoidFunction : public GradFunctionImpl<SigmoidFunction> {
50public:
51 SigmoidFunction() {
52 m_op_type = OpType::Sigmoid;
53 m_name = "SigmoidFunction";
54 }
55
56 tensor::Tensor<float> forward(
57 const std::vector<tensor::Tensor<float>>& inputs,
58 AutogradContext& ctx) override {
59 BREZEL_ENSURE(inputs.size() == 1,
60 "SigmoidFunction expects 1 input, got {}", inputs.size());
61
62 auto input = inputs[0];
63 auto neg_one = tensor::Tensor<float>({1}, -1.0f);
64 auto output = tensor::Tensor<float>::ones(input.shape())
65 .divide(tensor::Tensor<float>::ones(input.shape()) +
66 (input.multiply(neg_one)).exp());
67
68 ctx.save_for_backward("output", output);
69 return output;
70 }

84 const auto& output = ctx.get_saved_tensor("output");
85
86 auto ones = tensor::Tensor<float>::ones(output.shape());
87 auto sigmoid_grad = output.multiply(ones.subtract(output));
88 auto grad_input = sigmoid_grad.multiply(grad_output);

97class BREZEL_API TanhFunction : public GradFunctionImpl<TanhFunction> {
98public:
99 TanhFunction() {
100 m_op_type = OpType::Tanh;
101 m_name = "TanhFunction";
102 }
103
104 tensor::Tensor<float> forward(
105 const std::vector<tensor::Tensor<float>>& inputs,
106 AutogradContext& ctx) override {
107 BREZEL_ENSURE(inputs.size() == 1,
108 "TanhFunction expects 1 input, got {}", inputs.size());
109
110 auto input = inputs[0];
111 auto pos_exp = input.exp();
112 auto neg_exp =
113 (input.multiply(tensor::Tensor<float>({1}, -1.0f))).exp();
114 auto output = (pos_exp - neg_exp).divide(pos_exp + neg_exp);
115
116 ctx.save_for_backward("output", output);
117 return output;
118 }
119
129 std::vector<tensor::Tensor<float>> backward_impl(
130 const tensor::Tensor<float>& grad_output,
131 const AutogradContext& ctx) override {
132 const auto& output = ctx.get_saved_tensor("output");
133
134 auto grad_input =
135 (tensor::Tensor<float>::ones(output.shape()) - (output * output)) *
136 grad_output;
137
138 return {grad_input};
139 }

145class BREZEL_API LeakyReLUFunction
146 : public GradFunctionImpl<LeakyReLUFunction> {
147public:
148 LeakyReLUFunction(float negative_slope = 0.01)
149 : m_negative_slope(negative_slope) {
150 m_op_type = OpType::ReLU;
151 m_name = "LeakyReLUFunction";
152 }
153
154 tensor::Tensor<float> forward(
155 const std::vector<tensor::Tensor<float>>& inputs,
156 AutogradContext& ctx) override {
157 BREZEL_ENSURE(inputs.size() == 1,
158 "LeakyReLUFunction expects 1 input, got {}",
159 inputs.size());
160
161 auto input = inputs[0];
162 auto positive_part =
163 input.clip(0.0f, std::numeric_limits<float>::max());
164 auto negative_part =
165 input.clip(std::numeric_limits<float>::lowest(), 0.0f)
166 .multiply(tensor::Tensor<float>({1}, m_negative_slope));
167 auto output = positive_part + negative_part;
168
169 ctx.save_for_backward("input", input);
170 ctx.save_for_backward("slope",
171 static_cast<int64_t>(static_cast<int64_t>(
172 std::round(m_negative_slope * 1000.0f))));
173
174 return output;
175 }
176
177 std::vector<tensor::Tensor<float>> backward_impl(
178 const tensor::Tensor<float>& grad_output,
179 const AutogradContext& ctx) override {
180 const auto& input = ctx.get_saved_tensor("input");
181 float slope =
182 static_cast<float>(ctx.get_saved_scalar("slope")) / 1000.0f;
183
184 auto pos_mask = input.sign().clip(0.0f, 1.0f);
185 auto neg_mask = tensor::Tensor<float>::ones(input.shape()) - pos_mask;
186
187 auto slope_tensor = tensor::Tensor<float>({1}, slope);
188 auto grad_mask = pos_mask + neg_mask * slope_tensor;
189
190 return {grad_mask * grad_output};
191 }
192
193private:
194 float m_negative_slope;
195};

201class BREZEL_API ELUFunction : public GradFunctionImpl<ELUFunction> {
202public:
203 ELUFunction(float alpha = 1.0) : m_alpha(alpha) {
204 m_op_type = OpType::None; // Custom activation
205 m_name = "ELUFunction";
206 }
207
208 tensor::Tensor<float> forward(
209 const std::vector<tensor::Tensor<float>>& inputs,
210 AutogradContext& ctx) override {
211 BREZEL_ENSURE(inputs.size() == 1, "ELUFunction expects 1 input, got {}",
212 inputs.size());
213
214 auto input = inputs[0];
215 auto positive_part =
216 input.clip(0.0f, std::numeric_limits<float>::max());
217 auto negative_part =
218 (input.clip(std::numeric_limits<float>::lowest(), 0.0f).exp() -
219 tensor::Tensor<float>({1}, 1.0f));
220 auto output = positive_part + negative_part;
221
222 ctx.save_for_backward("output", output);
223 ctx.save_for_backward("input", input);
224 ctx.save_for_backward(
225 "alpha", static_cast<int64_t>(
226 static_cast<int64_t>(std::round(m_alpha * 1000.0f))));
227
228 return output;
229 }
230
240 std::vector<tensor::Tensor<float>> backward_impl(
241 const tensor::Tensor<float>& grad_output,
242 const AutogradContext& ctx) override {
243 const auto& output = ctx.get_saved_tensor("output");
244 const auto& input = ctx.get_saved_tensor("input");
245 float alpha =
246 static_cast<float>(ctx.get_saved_scalar("alpha")) / 1000.0f;
247
248 auto pos_mask = input.sign().clip(0.0f, 1.0f);
249 auto neg_mask = tensor::Tensor<float>::ones(input.shape()) - pos_mask;
250
251 auto pos_grad = pos_mask;
252 auto alpha_tensor = tensor::Tensor<float>({1}, alpha);
253 auto neg_grad = neg_mask * (output + alpha_tensor);
254
255 auto grad_mask = pos_grad + neg_grad;
256
257 return {grad_mask * grad_output};
258 }
259
260private:
261 float m_alpha;
262};
