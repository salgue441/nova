203 BREZEL_NODISCARD static BroadcastPlan create_broadcast_plan(
204 const std::vector<Shape>& shapes, const std::vector<Strides>& strides) {
205 if (shapes.empty() || strides.empty() ||
206 shapes.size() != strides.size()) {
207 throw core::error::InvalidArgument(
208 "Invalid input for broadcast plan creation");
209 }
210
211 Shape broadcasted_shape = broadcast_shapes(shapes);
212
213 BroadcastPlan plan;
214 plan.broadcasted_shape = broadcasted_shape;
215 plan.total_elements = broadcasted_shape.numel();
216
217 plan.dimensions.reserve(broadcasted_shape.size());
218
219 for (size_t dim = 0; dim < broadcasted_shape.size(); ++dim) {
220 BroadcastDimension dimension;
221 dimension.size = broadcasted_shape[dim];
222 dimension.strides.resize(shapes.size());
223
224 bool is_contiguous = true;
225
226 for (size_t tensor_idx = 0; tensor_idx < shapes.size();
227 ++tensor_idx) {
228 if (dim >=
229 broadcasted_shape.size() - shapes[tensor_idx].size()) {
230 size_t original_dim = dim - (broadcasted_shape.size() -
231 shapes[tensor_idx].size());
232
233 if (shapes[tensor_idx][original_dim] == 1) {
234 dimension.strides[tensor_idx] = 0;
235 is_contiguous = false;
236 } else {
237 dimension.strides[tensor_idx] =
238 strides[tensor_idx][original_dim];
239 }
240 } else {
241 dimension.strides[tensor_idx] = 0;
242 is_contiguous = false;
243 }
244 }
245
246 dimension.is_contiguous = is_contiguous;
247 plan.dimensions.push_back(dimension);
248 }
249
250 return plan;
251 }

264 BREZEL_NODISCARD static BroadcastPlan optimize_broadcast_plan(
265 const BroadcastPlan& plan) {
266 if (plan.dimensions.empty()) {
267 return plan;
268 }
269
270 BroadcastPlan optimized;
271 optimized.broadcasted_shape = plan.broadcasted_shape;
272 optimized.total_elements = plan.total_elements;
273
274 size_t num_tensors = plan.dimensions[0].strides.size();
275
276 size_t current_dim = 0;
277 while (current_dim < plan.dimensions.size()) {
278 BroadcastDimension merged = plan.dimensions[current_dim];
279 size_t next_dim = current_dim + 1;
280
281 while (next_dim < plan.dimensions.size()) {
282 const BroadcastDimension& next = plan.dimensions[next_dim];
283
284 bool can_merge = true;
285 for (size_t i = 0; i < num_tensors; ++i) {
286 if (!((merged.strides[i] == 0 && next.strides[i] == 0) ||
287 (merged.strides[i] != 0 && next.strides[i] != 0 &&
288 merged.strides[i] * merged.size ==
289 next.strides[i]))) {
290 can_merge = false;
291 break;
292 }
293 }
294
295 if (!can_merge) {
296 break;
297 }
298
299 for (size_t i = 0; i < num_tensors; ++i) {
300 if (merged.strides[i] == 0) {
301 merged.strides[i] = 0;
302 }
303 }
304
305 merged.size *= next.size;
306 merged.is_contiguous =
307 merged.is_contiguous && next.is_contiguous;
308
309 next_dim++;
310 }
311
312 optimized.dimensions.push_back(merged);
313 current_dim = next_dim;
314 }
315
316 return optimized;
317 }

332 template <typename Func, typename... DataPtrs>
333 static void apply_broadcasted(Func&& func, const std::vector<Shape>& shapes,
334 const std::vector<Strides>& strides,
335 DataPtrs... data_ptrs) {
336 static_assert(sizeof...(DataPtrs) > 0,
337 "At least one data pointer required");
338
339 if (shapes.empty() || shapes.size() != sizeof...(DataPtrs)) {
340 throw core::error::InvalidArgument(
341 "Number of shapes must match number of data pointers");
342 }
343
344 if (strides.size() != shapes.size()) {
345 throw core::error::InvalidArgument(
346 "Number of strides must match number of shapes");
347 }
348
349 BroadcastPlan plan = create_broadcast_plan(shapes, strides);
350 plan = optimize_broadcast_plan(plan);
351
352 apply_broadcast_plan(
353 std::forward<Func>(func), plan, 0,
354 std::vector<int64_t>(plan.broadcasted_shape.size(), 0), 0,
355 data_ptrs...);
356 }

372 template <typename Func, typename... DataPtrs>
373 static void apply_broadcast_plan(Func&& func, const BroadcastPlan& plan,
374 size_t dim, std::vector<int64_t>& indices,
375 size_t offset, DataPtrs... data_ptrs) {
376 if (dim == plan.dimensions.size()) {
377 func(data_ptrs...);
378 return;
379 }
380
381 const BroadcastDimension& dimension = plan.dimensions[dim];
382
383 std::array<int64_t, sizeof...(DataPtrs)> strides{
384 plan.dimensions[dim].strides[static_cast<size_t>(
385 &data_ptrs - &std::get<0>(std::make_tuple(data_ptrs...)))]...};
386
387 for (size_t i = 0; i < dimension.size; ++i) {
388 indices[dim] = i;
389
390 if (dim == plan.dimensions.size() - 1) {
391 func((data_ptrs +
392 (strides[&data_ptrs -
393 &std::get<0>(std::make_tuple(data_ptrs...))] *
394 i))...);
395 } else {
396 apply_broadcast_plan(
397 std::forward<Func>(func), plan, dim + 1, indices, offset,
398 (data_ptrs +
399 (strides[&data_ptrs -
400 &std::get<0>(std::make_tuple(data_ptrs...))] *
401 i))...);
402 }
403 }
