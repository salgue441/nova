171 const FunctionType& get_function() const {
172 auto key = std::make_pair(std::type_index(typeid(T1)),
173 std::type_index(typeid(T2)));
174
175 auto it = m_functions.find(key);
176
177 if (it == m_functions.end()) {
178 throw core::error::LogicError(
179 "No function registered for types {} and {}", typeid(T1).name(),
180 typeid(T2).name());

194 bool has_function() const {

203 struct KeyHash {
204 std::size_t operator()(const KeyType& key) const {
205 return std::hash<std::type_index>()(key.first) ^
206 std::hash<std::type_index>()(key.second);
207 }
208 };

210 struct KeyEqual {

212 return lhs.first == rhs.first && lhs.second == rhs.second;

919template <typename Ret, typename Arg1, typename Arg2, typename... Args>
920Ret dispatch_binary_function(
921 const BinaryTypeDispatchRegistry<Ret, Arg1, Arg2, Args...>& registry,
922 DataType dtype1, DataType dtype2, Arg1 arg1, Arg2 arg2, Args... args) {
923#define DISPATCH_BINARY(T1, T2) \
924 return registry.template get_function<T1, T2>()( \
925 arg1, arg2, std::forward<Args>(args)...)
926
927 if (dtype1 == DataType::Float32 && dtype2 == DataType::Float32) {
928 DISPATCH_BINARY(float, float);
929 } else if (dtype1 == DataType::Float64 && dtype2 == DataType::Float64) {
930 DISPATCH_BINARY(double, double);
931 } else if (dtype1 == DataType::Int32 && dtype2 == DataType::Int32) {
932 DISPATCH_BINARY(int32_t, int32_t);
933 } else if (dtype1 == DataType::Int64 && dtype2 == DataType::Int64) {
934 DISPATCH_BINARY(int64_t, int64_t);
935 }
936
937 switch (dtype1) {
938 case DataType::Float32:
939 switch (dtype2) {
940 case DataType::Float32:
941 DISPATCH_BINARY(float, float);
942 case DataType::Float64:
943 DISPATCH_BINARY(float, double);
944 case DataType::Int32:
945 DISPATCH_BINARY(float, int32_t);
946 case DataType::Int64:
947 DISPATCH_BINARY(float, int64_t);
948 case DataType::Complex64:
949 DISPATCH_BINARY(float, std::complex<float>);
950 default:
951 break;
952 }
953 break;
954
955 case DataType::Float64:
956 switch (dtype2) {
957 case DataType::Float32:
958 DISPATCH_BINARY(double, float);
959 case DataType::Float64:
960 DISPATCH_BINARY(double, double);
961 case DataType::Int32:
962 DISPATCH_BINARY(double, int32_t);
963 case DataType::Int64:
964 DISPATCH_BINARY(double, int64_t);
965 case DataType::Complex128:
966 DISPATCH_BINARY(double, std::complex<double>);
967 default:
968 break;
969 }
970 break;
971
972 case DataType::Int32:
973 switch (dtype2) {
974 case DataType::Float32:
975 DISPATCH_BINARY(int32_t, float);
