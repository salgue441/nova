82inline void check_matmul_shapes(const Shape& lhs, const Shape& rhs) {
83 if (lhs.size() < 1 || rhs.size() < 1) {
84 throw core::error::LogicError(
85 "Both tensors must have at least 1 dimension for matmul");
86 }
87
88 const int64_t lhs_dim = lhs.size() >= 2 ? lhs[lhs.size() - 1] : lhs[0];
89 const int64_t rhs_dim = rhs.size() >= 2 ? rhs[rhs.size() - 2] : rhs[0];
90
91 if (lhs_dim != rhs_dim) {
92 throw core::error::LogicError(
93 "Incompatible dimensions for matrix multiplication: {} and {}",
94 lhs.to_string(), rhs.to_string());
95 }

201inline Shape check_reduction_shape(const Shape& shape, int64_t dim,
202 bool keepdim = false) {
203 if (dim == -1) {
204 if (keepdim) {
205 Shape result;
206
207 for (size_t i = 0; i < shape.size(); ++i) {
208 result.push_back(1);
209 }
210
211 return result;
212 } else {
213 return Shape({1});
214 }
215 }
216
217 if (dim < 0) {
218 dim += shape.size();
219 }
220
221 if (dim < 0 || dim >= static_cast<int64_t>(shape.size())) {
222 throw core::error::LogicError(
223 "Invalid reduction dimension {} for shape with {} dimensions", dim,
224 shape.size());
225 }
226
227 Shape result;
228 for (size_t i = 0; i < shape.size(); ++i) {
229 if (i == static_cast<size_t>(dim)) {
230 if (keepdim) {
231 result.push_back(1);
232 }
233 } else {
234 result.push_back(shape[i]);
235 }
236 }
237
238 return result;
239}
