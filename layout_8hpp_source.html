97constexpr int cound_leading_zeros(uint64_t x) noexcept {
98#if defined(__GNUC__) || defined(__clang__)
99 return x ? __builtin_clzll(x) : 64;
100#elif defined(_MSC_VER)
101 unsigned long result;
102 return _BitScanReverse64(&result, x) ? 63 - result : 64;
103#else
104 if (x == 0)
105 return;

109 count++;
110 x <<= 1;

308 BREZEL_FORCE_INLINE bool next() noexcept {
309 if (++m_position >= m_total_size) {
310 return false;
311 }
312
313 for (int64_t dim = m_ndim - 1; dim >= 0; --dim) {
314 if (++m_coordinates[dim] < m_shape[dim]) {
315 m_offset += m_strides[dim];
316 return true;
317 }

319 m_coordinates[dim] = 0;
320 m_offset -= m_strides[dim] * (m_shape[dim] - 1);

342 BREZEL_FORCE_INLINE void move_to(std::span<const int64_t> coords) {
343 if (coords.size() != m_ndim) {

531 BREZEL_NODISCARD LayoutDescriptor(const Shape& shape, MemoryFormat format,
532 DeviceType device = DeviceType::CPU)
533 : m_shape(shape), m_format(format), m_device(device) {
534 validate_shape(shape);
535
536 if (format == MemoryFormat::ChannelsLast) {
537 if (shape.size() == 4) {
538 m_layout = MemoryLayout::Strided;
539
540 // Compute NHWC strides (N x H x W x C)
541 int64_t h = shape[1];
542 int64_t w = shape[2];
543 int64_t c = shape[3];
544
545 m_strides.resize(4);
546 m_strides[0] = h * w * c; // N stride
547 m_strides[1] = w * c; // H stride
548 m_strides[2] = c; // W stride
549 m_strides[3] = 1; // C stride
550 } else {
551 throw brezel::core::error::InvalidArgument(
552 "ChannelsLast format requires a 4D tensor, but got {}D",
553 shape.size());
554 }
555 } else if (format == MemoryFormat::ChannelsFirst) {
556 if (shape.size() == 4) {
557 m_layout = MemoryLayout::RowMajor;
558 compute_strides();
559 } else {
560 throw brezel::core::error::InvalidArgument(
561 "ChannelsFirst format requires a 4D tensor, but got {}D",
562 shape.size());
563 }
564 } else if (format == MemoryFormat::Sparse) {
565 m_layout = MemoryLayout::Strided;
566 m_strides.resize(shape.size(), 0);
567 } else {
568 m_layout = MemoryLayout::RowMajor;
569 compute_strides();
570 }
571 }

649 BREZEL_NODISCARD BREZEL_FORCE_INLINE size_t
650 get_linear_index(std::span<const int64_t> indices) const {
651 if (detail::unlikely(indices.size() != m_shape.size())) {
652 throw brezel::core::error::InvalidArgument(
653 "Indices size must match shape size");
654 }
655
656#ifdef BREZEL_CONFIG_DEBUG
657 for (size_t i = 0; i < indices.size(); ++i) {
658 if (detail::unlikely(indices[i] < 0 || indices[i] >= m_shape[i])) {
659 throw brezel::core::error::LogicError("Index out of bounds");
660 }
661 }
662#endif
663
664 size_t offset = m_offset;
665 const size_t ndim = m_shape.size();
666
667 // Optimization for common tensor dimensions with unrolled loops
668 if (ndim <= 4) {
669 switch (ndim) {
670 case 0:
671 return offset;
672
673 case 1:
674 return offset + indices[0] * m_strides[0];
675
676 case 2:
677 return offset + indices[0] * m_strides[0] +
678 indices[1] * m_strides[1];
679
680 case 3:
681 return offset + indices[0] * m_strides[0] +
682 indices[1] * m_strides[1] +
683 indices[2] * m_strides[2];
684
685 case 4:
686 return offset + indices[0] * m_strides[0] +
687 indices[1] * m_strides[1] +
688 indices[2] * m_strides[2] +
689 indices[3] * m_strides[3];
690 }
691 }
692
693 for (size_t i = 0; i < ndim; ++i) {
694 offset += indices[i] * m_strides[i];
695 }
696
697 return offset;
698 }

707 BREZEL_NODISCARD BREZEL_FORCE_INLINE size_t simd_dot_product(
708 std::span<const int64_t> a, std::span<const int64_t> b) const noexcept {
709 size_t result = 0;
710
711#if defined(BREZEL_SIMD_AVX512)
712 if (a.size() >= 8) {
713 const size_t simd_size = a.size() - (a.size() % 8);
714 for (size_t i = 0; i < simd_size; i += 8) {
715 __m512i a_vec = _mm512_loadu_epi64(a.data() + i);
716 __m512i b_vec = _mm512_loadu_epi64(b.data() + i);
717 __m512i prod = _mm512_mullo_epi64(a_vec, b_vec);
718 result += _mm512_reduce_add_epi64(prod);
719 }
720
721 // Process remaining elements
722 for (size_t i = simd_size; i < a.size(); ++i) {
723 result += a[i] * b[i];
724 }
725 } else {
726 result =
727 std::inner_product(a.begin(), a.end(), b.begin(), size_t{0});
728 }
729#elif defined(BREZEL_SIMD_AVX2)
730 if (a.size() >= 4) {
731 const size_t simd_size = a.size() - (a.size() % 4);
732
733 for (size_t i = 0; i < simd_size; i += 4) {
734 __m256i a_vec = _mm256_loadu_si256(
735 reinterpret_cast<const __m256i*>(a.data() + i));
736 __m256i b_vec = _mm256_loadu_si256(
737 reinterpret_cast<const __m256i*>(b.data() + i));
738
739 __m256i prod = _mm256_mul_epi32(a_vec, b_vec);
740
741 int64_t sum_array[4];
742 _mm256_storeu_si256(reinterpret_cast<__m256i*>(sum_array),
743 prod);
744
745 for (int j = 0; j < 4; ++j) {
746 result += sum_array[j];
747 }
748 }
749
750 // Process remaining elements
751 for (size_t i = simd_size; i < a.size(); ++i) {
752 result += a[i] * b[i];
753 }
754 } else {
755 result =
756 std::inner_product(a.begin(), a.end(), b.begin(), size_t{0});
757 }
758#else
759 // No SIMD support, std
760 result = std::inner_product(a.begin(), a.end(), b.begin(), size_t{0});
761#endif
762
763 return result;
764 }

774 BREZEL_FORCE_INLINE void get_indices(size_t linear_index,
775 std::span<int64_t> indices) const {
776 if (detail::unlikely(indices.size() != m_shape.size())) {
777 throw brezel::core::error::InvalidArgument(
778 "Output indices size ({}) must match shape size ({})",
779 indices.size(), m_shape.size());
780 }
781
782 size_t remaining = linear_index - m_offset;
783 if (m_layout == MemoryLayout::RowMajor) {
784 const size_t ndim = m_shape.size();
785
786 if (ndim <= 4) {
787 switch (ndim) {
788 case 0:
789 return;
790
791 case 1:
792 indices[0] = remaining / m_strides[0];
793 return;
794
795 case 2:
796 indices[1] = remaining % m_strides[0];
797 indices[0] = remaining / m_strides[0];
798 return;
799
800 case 3:
801 indices[2] = remaining % m_strides[1];
802 remaining /= m_strides[1];
803 indices[1] = remaining % (m_strides[0] / m_strides[1]);
804 indices[0] = remaining / (m_strides[0] / m_strides[1]);
805 return;
806
807 case 4:
808 indices[3] = remaining % m_strides[2];
809 remaining /= m_strides[2];
810 indices[2] = remaining % (m_strides[1] / m_strides[2]);
811 remaining /= (m_strides[1] / m_strides[2]);
812 indices[1] = remaining % (m_strides[0] / m_strides[1]);
813 indices[0] = remaining / (m_strides[0] / m_strides[1]);
814 return;
815 }
816 }
817
818 for (size_t i = 0; i < ndim; ++i) {
819 indices[i] = remaining / m_strides[i];
820 remaining %= m_strides[i];
821 }
822
823 return;
824 }
825
826 if (m_layout == MemoryLayout::ColumnMajor) {
827 const size_t ndim = m_shape.size();
828
829 if (ndim <= 4) {
830 switch (ndim) {
831 case 0:
832 return;
833
834 case 1:
835 indices[0] = remaining / m_strides[0];
836 return;
837
838 case 2:
839 indices[0] = remaining % m_strides[1];
840 indices[1] = remaining / m_strides[1];
841 return;
842
843 case 3:
844 indices[0] = remaining % m_strides[1];
845 remaining /= m_strides[1];
846 indices[1] = remaining % (m_strides[2] / m_strides[1]);
847 indices[2] = remaining / (m_strides[2] / m_strides[1]);
848 return;
849
850 case 4:
851 indices[0] = remaining % m_strides[1];
852 remaining /= m_strides[1];
853 indices[1] = remaining % (m_strides[2] / m_strides[1]);
854 remaining /= (m_strides[2] / m_strides[1]);
855 indices[2] = remaining % (m_strides[3] / m_strides[2]);
856 indices[3] = remaining / (m_strides[3] / m_strides[2]);
857 return;
858 }
859 }
860
861 for (int64_t i = ndim - 1; i >= 0; --i) {
862 indices[i] = remaining / m_strides[i];
863 remaining %= m_strides[i];
864 }
865 return;
866 }
867
868 compute_indices_from_offset(remaining, indices);
869 }

877 BREZEL_FORCE_INLINE void compute_indices_from_offset(
878 size_t offset, std::span<int64_t> indices) const {
879 std::fill(indices.begin(), indices.end(), 0);
880 if (offset == 0) {
881 return;
882 }
883
884 const size_t ndim = m_shape.size();
885 if (ndim == 0) {
886 return;
887 }
888
889 size_t remaining = offset;
890 bool progress = true;
891
892 // Greedy algorithm for computing indices
893 while (remaining > 0 && progress) {
894 progress = false;
895
896 for (size_t i = 0; i < ndim; ++i) {
897 if (m_strides[i] > 0 && indices[i] < m_shape[i] - 1 &&
898 remaining >= m_strides[i]) {
899 remaining -= m_strides[i];
900 indices[i]++;
901
902 progress = true;
903 }
904 }
905 }
906 }

1076 BREZEL_NODISCARD LayoutDescriptor transpose(size_t dim0,
1077 size_t dim1) const {
1078 if (dim0 >= m_shape.size() || dim1 >= m_shape.size()) {
1079 throw brezel::core::error::InvalidArgument(
1080 "Transposition dimensions ({}, {}) out of bounds for shape of "
1081 "size {}",
1082 dim0, dim1, m_shape.size());
1083 }
1084
1085 if (dim0 == dim1) {
1086 return *this;
1087 }
1088
1089 Shape new_shape = m_shape;
1090 std::swap(new_shape[dim0], new_shape[dim1]);
1091
1092 auto new_strides = m_strides;
1093 std::swap(new_strides[dim0], new_strides[dim1]);
1094
1095 auto result = LayoutDescriptor(new_shape, new_strides);
1096 result.set_offset(m_offset);
1097 result.set_format(m_format);
1098 result.set_device(m_device);
1099
1100 return result;

1109 LayoutDescriptor permute(const std::vector<size_t>& dims) const {
1110 if (dims.size() != m_shape.size()) {
1111 throw core::error::InvalidArgument(
1112 "Permutation dimensions ({}) must match shape size ({})",
1113 dims.size(), m_shape.size());
1114 }
1115
1116 std::vector<bool> used(m_shape.size(), false);
1117 for (size_t dim : dims) {
1118 if (dim >= m_shape.size()) {
1119 throw core::error::InvalidArgument(
1120 "Dimension {} is out of bounds for shape with {} "
1121 "dimensions",
1122 dim, m_shape.size());
1123 }
1124
1125 if (used[dim]) {
1126 throw core::error::InvalidArgument(
1127 "Duplicate dimension {} in permutation", dim);
1128 }
1129
1130 used[dim] = true;
1131 }
1132
1133 Shape new_shape;
1134 for (size_t dim : dims) {
1135 new_shape.push_back(m_shape[dim]);
1136 }
1137
1138 std::vector<int64_t> new_strides(dims.size());
