125class BREZEL_API DefaultAllocator : public IAllocator {
126public:
127 // Constructor
128 DefaultAllocator() = default;

130 // Destructor to ensure proper cleanup
131 ~DefaultAllocator() {
132 // Clean up any remaining allocations
133 for (auto& ptr : m_aligned_allocations) {
134#if defined(BREZEL_PLATFORM_WINDOWS)
135 _aligned_free(ptr);
136#else
137 free(ptr);
138#endif
139 }
140 }

150 BREZEL_NODISCARD void* allocate(
151 size_t size, Alignment alignment = Alignment::Default,
152 AllocFlags flags = AllocFlags::None) override {
153 if (size == 0)
154 return nullptr;
155
156 void* ptr = nullptr;
157 const bool zero_memory =
158 (flags & AllocFlags::ZeroMemory) == AllocFlags::ZeroMemory;
159
160 if (alignment == Alignment::Default) {
161 ptr = tbb::scalable_allocator<std::byte>().allocate(size);
162 } else {
163 const size_t align_val = static_cast<size_t>(alignment);
164
165#if defined(BREZEL_PLATFORM_WINDOWS)
166 ptr = _aligned_malloc(size, align_val);
167#else
168 if (posix_memalign(&ptr, align_val, size) != 0) {
169 throw brezel::core::error::RuntimeError(
170 "Failed to allocate {} bytes with alignment {}", size,
171 align_val);
172 }
173#endif
174
175 // Track aligned allocations
176 if (ptr) {
177 std::lock_guard<std::mutex> lock(m_mutex);
178 m_aligned_allocations.insert(ptr);
179 }
180 }
181
182 if (!ptr) {
183 throw brezel::core::error::RuntimeError(
184 "Memory allocation failed for {} bytes", size);
185 }
186
187 if (zero_memory) {
188 std::memset(ptr, 0, size);
189 }
190
191 {
192 std::lock_guard<std::mutex> lock(m_mutex);
193 m_allocated_bytes += size;
194 }
195
196 return ptr;
197 }
198
205 void deallocate(void* ptr, size_t size = 0) noexcept override {
206 if (!ptr)
207 return;
208
209 if (size > 0) {
210 std::lock_guard<std::mutex> lock(m_mutex);
211 m_allocated_bytes -= std::min(m_allocated_bytes, size);
212 }
213
214 // Check if this was an aligned allocation
215 {
216 std::lock_guard<std::mutex> lock(m_mutex);
217 auto it = m_aligned_allocations.find(ptr);
218 if (it != m_aligned_allocations.end()) {
219#if defined(BREZEL_PLATFORM_WINDOWS)
220 _aligned_free(ptr);
221#else
222 free(ptr);
223#endif
224 m_aligned_allocations.erase(it);
225 return;
226 }
227 }
228
229 // Use TBB deallocator for regular allocations
230 tbb::scalable_allocator<std::byte>().deallocate(
231 static_cast<std::byte*>(ptr), size);
232 }
233
238 BREZEL_NODISCARD size_t get_allocated_bytes() const noexcept override {
239 std::lock_guard<std::mutex> lock(m_mutex);
240 return m_allocated_bytes;
241 }
242
243private:
244 mutable std::mutex m_mutex;
245 size_t m_allocated_bytes = 0;
246 std::unordered_set<void*> m_aligned_allocations;
247};

254class BREZEL_API PoolAllocator : public IAllocator {
255public:
263 explicit PoolAllocator(size_t block_size, size_t initial_capacity = 64,
264 Alignment alignment = Alignment::Default)
265 : m_block_size(block_size), m_alignment(alignment) {
266 grow_pool(initial_capacity);
267 }
268
269 ~PoolAllocator() {
270 for (auto& block : m_memory_blocks) {
271#if defined(BREZEL_PLATFORM_WINDOWS)
272 _aligned_free(block);
273#else
274 free(block);
275#endif
276 }
277 }
278
288 BREZEL_NODISCARD void* allocate(
289 size_t size, Alignment alignment = Alignment::Default,
290 AllocFlags flags = AllocFlags::None) override {
291 if (size > m_block_size) {
292 throw brezel::core::error::RuntimeError(
293 "Requested allocation size {} exceeds pool block size {}", size,
294 m_block_size);
295 }
296
297 std::lock_guard<std::mutex> lock(m_mutex);
298 if (m_free_list.empty()) {
299 grow_pool(m_memory_blocks.size() * 2);
300 }
301
302 void* ptr = m_free_list.back();
303 m_free_list.pop_back();
304
305 if ((flags & AllocFlags::ZeroMemory) == AllocFlags::ZeroMemory) {
306 std::memset(ptr, 0, m_block_size);
307 }
308
309 m_allocated_blocks++;
310 return ptr;
311 }
312
319 void deallocate(void* ptr, size_t size = 0) noexcept override {
320 if (!ptr)
321 return;
322
323 std::lock_guard<std::mutex> lock(m_mutex);
324 bool found = false;
325
326 for (auto& block : m_memory_blocks) {
327 uintptr_t block_addr = reinterpret_cast<uintptr_t>(block);
328 uintptr_t ptr_addr = reinterpret_cast<uintptr_t>(ptr);
329
330 if (ptr_addr >= block_addr &&
331 ptr_addr < block_addr + m_blocks_per_chunk * m_block_size) {
332 found = true;
333 break;
334 }
335 }
336
337 if (!found)
338 return;
339
340 m_free_list.push_back(ptr);
341 m_allocated_blocks--;
342 }
343
348 BREZEL_NODISCARD bool supports_resize() const noexcept override {
349 return false;
350 }
351
356 BREZEL_NODISCARD size_t get_allocated_bytes() const noexcept override {
357 std::lock_guard<std::mutex> lock(m_mutex);
358 return m_allocated_blocks * m_block_size;
359 }
360
361private:
367 void grow_pool(size_t additional_blocks) {
368 const size_t min_blocks = 16;
369 additional_blocks = std::max(additional_blocks, min_blocks);
370
371 const size_t align_val = static_cast<size_t>(m_alignment);
372 const size_t chunk_size = additional_blocks * m_block_size;
373 void* chunk = nullptr;
374
375#if defined(BREZEL_PLATFORM_WINDOWS)
376 chunk = _aligned_malloc(chunk_size, align_val);
377#else
378 if (posix_memalign(&chunk, align_val, chunk_size) != 0) {
379 throw brezel::core::error::RuntimeError(
380 "Failed to allocate chunk of {} bytes with alignment {}",
381 chunk_size, align_val);
382 }
383#endif
384
385 if (!chunk) {
386 throw brezel::core::error::RuntimeError(
387 "Failed to grow memory pool by {} blocks", additional_blocks);
388 }
389
390 m_memory_blocks.push_back(chunk);
391 auto* current = static_cast<std::byte*>(chunk);
392
393 for (size_t i = 0; i < additional_blocks; ++i) {
394 m_free_list.push_back(current);
395 current += m_block_size;
396 }
397
398 m_blocks_per_chunk = additional_blocks;
399 }
400
401 size_t m_block_size;
402 Alignment m_alignment;
403 std::vector<void*> m_memory_blocks;
404 std::vector<void*> m_free_list;
405 size_t m_blocks_per_chunk = 0;
406 size_t m_allocated_blocks = 0;
407 mutable std::mutex m_mutex;
408};

439 BREZEL_NODISCARD IAllocator& get_pool_allocator(size_t block_size) {
440 std::lock_guard<std::mutex> lock(m_mutex);
441 auto it = m_pool_allocators.find(block_size);
442
443 if (it == m_pool_allocators.end()) {
444 auto [new_it, inserted] = m_pool_allocators.emplace(

641template <typename T>
642BREZEL_NODISCARD AllocatedPtr<T> make_allocated_array(
643 size_t count,
644 IAllocator& allocator = MemoryManager::instance().default_allocator(),
645 Alignment alignment = Alignment::Default, const T& init_val = T()) {
646 if (count == 0)
647 return AllocatedPtr<T>();
648
649 const size_t total_size = sizeof(T) * count;
650 void* memory = allocator.allocate(total_size, alignment);
651 T* array = static_cast<T*>(memory);
652
653 if constexpr (std::is_trivially_constructible_v<T>) {
654 T default_val = T();
655 if (std::memcmp(&init_val, &default_val, sizeof(T)) == 0) {
656 std::memset(array, 0, total_size);
657 } else {
658 for (size_t i = 0; i < count; ++i) {
659 std::memcpy(&array[i], &init_val, sizeof(T));
660 }
661 }
662 } else {
663 for (size_t i = 0; i < count; ++i) {
664 new (&array[i]) T(init_val);
665 }
666 }
667
668 return AllocatedPtr<T>(array, &allocator, total_size);
669}
