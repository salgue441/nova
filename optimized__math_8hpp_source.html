44template <typename T>
45BREZEL_NODISCARD BREZEL_FORCE_INLINE T fast_exp(T x) {
46 static_assert(std::is_floating_point_v<T>,
47 "fast_exp requires floating point type");
48
49 if (x < -87.0f) {
50 return 0.0f;
51 }
52
53 if (x > 88.0f) {
54 return std::numeric_limits<float>::infinity();
55 }
56
57 const float ln2 = 0.6931471805599453f;
58 const float half_ln2 = ln2 * 0.5f;
59
60 float f = std::floor(x / ln2 + 0.5f);
61 float y = x - f * ln2;
62 float p =
63 1.0f +
64 y * (1.0f +
65 y * (0.5f + y * (0.166666667f +
66 y * (0.041666667f +
67 y * (0.008333333f + y * 0.001388889f)))));
68
69 int exponent = static_cast<int>(f);
70 return std::ldexp(static_cast<T>(p), exponent);
71}

93template <typename T>
94BREZEL_NODISCARD BREZEL_FORCE_INLINE T fast_log(T x) {
95 static_assert(std::is_floating_point_v<T>,
96 "fast_log requires floating point type");
97
98 if (x <= 0) {
99 return -std::numeric_limits<T>::infinity();
100 }
101
102 int exponent;
103 float mantissa = std::frexp(x, &exponent);
104
105 if (mantissa < 1.0f) {
106 mantissa *= 2.0f;
107 exponent--;
108 }
109
110 float y = (mantissa - 1.0f) / (mantissa + 1.0f);
111 float y2 = y * y;
112 float log_mantissa =
113 y * (2.0f + y2 * (2.0f / 3.0f + y2 * (2.0f / 5.0f + y2 * 2.0f / 7.0f)));
114
115 const float ln2 = 0.6931471805599453f;
116 return static_cast<T>(log_mantissa + exponent * ln2);
117}

135template <typename T>
136BREZEL_NODISCARD BREZEL_FORCE_INLINE T fast_sqrt(T x) {
137 if constexpr (std::is_same_v<T, float>) {
138#if defined(BREZEL_SIMD_AVX2) || defined(BREZEL_SIMD_AVX512)
139 __m128 value = _mm_set_ss(x);
140 __m128 result = _mm_sqrt_ss(value);
141
142 return _mm_cvtss_f32(result);
143#else
144 union {
145 float f;
146 u_int32_t i;
147 } u;
148
149 u.f = x;
150 u.i = (u.i + 0x3f76cf62) >> 1;
151 u.f = 0.5f * (u.f + x / u.f);
152
153 return u.f;
154#endif
155 } else if constexpr (std::is_same_v<T, double>) {
156#if defined(BREZEL_SIMD_AVX2) || defined(BREZEL_SIMD_AVX512)
157 __m128d value = _mm_set_sd(x);
158 __m128d result = _mm_sqrt_sd(value, value);
159
160 return _mm_cvtsd_f64(result);
161#else
162 return std::sqrt(x);
163#endif } else {
164 return static_cast<T>(std::sqrt(static_cast<double>(x)));
165 }
166}

180template <typename T>
181BREZEL_NODISCARD BREZEL_FORCE_INLINE T fast_tanh(T x) {
182 static_assert(std::is_floating_point_v<T>,
183 "fast_tanh requires floating point type");
184
185 const float x2 = x * x;
186 if (x2 > 15.0f) {
187 return (x > 0.0f) ? T(1.0f) : T(-1.0f);
188 }
189
190 return static_cast<T>((x * (27.0f + x2)) / (27.0f + 9.0f * x2));
191}

285template <typename T>
286BREZEL_NODISCARD BREZEL_FORCE_INLINE T fast_pow(T base, int exponent) {
287 if (exponent == 0)
288 return T(1);
289
290 if (exponent == 1)
291 return base;
292
293 if (base == T(0))
294 return T(0);
295
296 bool is_negative = exponent < 0;
297 unsigned int abs_exp = std::abs(exponent);
298
299 T result = T(1);
300 T current_power = base;
301
302 while (abs_exp > 0) {
303 if (abs_exp & 1) {
304 result *= current_power;
305 }
306
307 current_power *= current_power;
308 abs_exp >>= 1;
309 }
310
311 return is_negative ? (T(1) / result) : result;
312}

339template <typename T>
340BREZEL_NODISCARD BREZEL_FORCE_INLINE T fast_pow(T base, T exponent) {
341 static_assert(std::is_floating_point_v<T>,
342 "Base type must be floating point");
343
344 if (base == T(0))
345 return T(0);
346
347 if (base == T(1))
348 return T(1);
349
350 if (exponent == T(0))
351 return T(1);
352
353 if (exponent == T(1))
354 return base;
355
356 if (std::floor(exponent) == exponent) {
357 return fast_pow(base, static_cast<int>(exponent));
358 }
359
360 return fast_exp(exponent * fast_log(base));
361}

383template <typename T>
384BREZEL_NODISCARD BREZEL_FORCE_INLINE T fast_abs(T x) {
385 if constexpr (std::is_floating_point_v<T>) {
386 if constexpr (std::is_same_v<T, float>) {
387 union {
388 float f;
389 uint32_t i;
390 } u{x};
391
392 // Clear sign bit
393 u.i &= 0x7FFFFFFF;
394 return u.f;
395 } else if constexpr (std::is_same_v<T, double>) {
396 union {
397 double d;
398 uint64_t i;
399 } u{x};
400
401 // Clear sign bit
402 u.i &= 0x7FFFFFFFFFFFFFFFULL;
403 return u.d;
404 }
405 } else if constexpr (std::is_integral_v<T> && std::is_signed_v<T>) {
406 const T mask = x >> (sizeof(T) * 8 - 1);
407 return (x + mask) ^ mask;
408 }
409
410 return std::abs(x);
411}

430template <typename T>
431BREZEL_NODISCARD BREZEL_FORCE_INLINE T fast_inv_sqrt(T x) {
432 static_assert(std::is_floating_point_v<T>,
433 "fast_inv_sqrt requires floating point type");
434
435 if constexpr (std::is_same_v<T, float>) {
436 union {
437 float f;
438 uint32_t i;
439 } u;
440
441 u.f = x;
442 u.i = 0x5F3759DF - (u.i >> 1);
443
444 float xhalf = 0.5f * x;
445 u.f = u.f * (1.5f - xhalf * u.f * u.f);
446
447 return u.f;
448 } else {
449 return 1.0 / std::sqrt(x);
450 }

484BREZEL_FORCE_INLINE void batch_exp_f32(float* data, size_t count) {
485#if defined(BREZEL_SIMD_AVX512)
486 size_t vec_count = count / 16;
487 size_t i = 0;
488
489 for (; i < vec_count * 16; i += 16) {
490 __m512 x = _mm512_load_ps(data + i);
491
492 // Apply the same algorithm as in fast_exp, but vectorized
493 __m512 mask_under =
494 _mm512_cmp_ps_mask(x, _mm512_set1_ps(-87.0f), _CMP_LT_OS);
495 __m512 mask_over =
496 _mm512_cmp_ps_mask(x, _mm512_set1_ps(88.0f), _CMP_GT_OS);
497
498 const __m512 ln2 = _mm512_set1_ps(0.6931471805599453f);
499 __m512 f = _mm512_floor_ps(_mm512_div_ps(x, ln2));
500 __m512 y = _mm512_sub_ps(x, _mm512_mul_ps(f, ln2));
501
502 // Polynomial approximation
503 __m512 p = _mm512_add_ps(
504 _mm512_set1_ps(1.0f),
505 _mm512_mul_ps(
506 y,
507 _mm512_add_ps(
508 _mm512_set1_ps(1.0f),
509 _mm512_mul_ps(
510 y,
511 _mm512_add_ps(
512 _mm512_set1_ps(0.5f),
513 _mm512_mul_ps(
514 y,
515 _mm512_add_ps(
516 _mm512_set1_ps(0.166666667f),
517 _mm512_mul_ps(
518 y,
519 _mm512_add_ps(
520 _mm512_set1_ps(0.041666667f),
521 _mm512_mul_ps(
522 y,
523 _mm512_add_ps(
524 _mm512_set1_ps(
525 0.008333333f),
526 _mm512_mul_ps(
527 y,
528 _mm512_set1_ps(
529 0.001388889f)))))))))))));
530
531 // 2^exponent using integer trick
532 __m512i exponent = _mm512_cvtps_epi32(f);
533 __m512i exp_shift = _mm512_slli_epi32(
534 _mm512_add_epi32(exponent, _mm512_set1_epi32(127)), 23);
535 __m512 exp_scale = _mm512_castsi512_ps(exp_shift);
536
537 __m512 result = _mm512_mul_ps(p, exp_scale);
538
539 // Handle underflow/overflow
540 result = _mm512_mask_blend_ps(mask_under, result, _mm512_setzero_ps());
541 result = _mm512_mask_blend_ps(
542 mask_over, result,
543 _mm512_set1_ps(std::numeric_limits<float>::infinity()));
544
545 _mm512_store_ps(data + i, result);
546 }
547

549 data[i] = fast_exp(data[i]);
550 }

569 _mm256_set1_ps(1.0f),
570 _mm256_mul_ps(
571 y,
572 _mm256_add_ps(
573 _mm256_set1_ps(1.0f),
574 _mm256_mul_ps(

653BREZEL_FORCE_INLINE void batch_log_f32(float* data, size_t count) {
654#if defined(BREZEL_SIMD_AVX512)
655 size_t vec_count = count / 16;
656 size_t i = 0;
657
658 for (; i < vec_count * 16; i += 16) {
659 __m512 x = _mm512_load_ps(data + i);
660
661 // Handle non-positive values
662 __mmask16 mask_nonpos =
663 _mm512_cmp_ps_mask(x, _mm512_setzero_ps(), _CMP_LE_OS);
664
665 // Extract mantissa and exponent using bit manipulation
666 __m512i xi = _mm512_castps_si512(x);
667 __m512i exponent = _mm512_srli_epi32(
668 _mm512_and_si512(xi, _mm512_set1_epi32(0x7F800000)), 23);
669 __m512i mantissa_bits =
670 _mm512_and_si512(xi, _mm512_set1_epi32(0x007FFFFF));
671 __m512 mantissa = _mm512_or_ps(_mm512_castsi512_ps(mantissa_bits),
672 _mm512_set1_ps(1.0f));
673
674 // Scale mantissa to [1, 2) range
675 __mmask16 scale_mask =
676 _mm512_cmp_ps_mask(mantissa, _mm512_set1_ps(1.0f), _CMP_LT_OS);
677 mantissa = _mm512_mask_mul_ps(mantissa, scale_mask, mantissa,
678 _mm512_set1_ps(2.0f));
679 __m512i exponent_adj = _mm512_mask_sub_epi32(
680 exponent, scale_mask, exponent, _mm512_set1_epi32(1));
681
682 // y = (mantissa - 1) / (mantissa + 1)
683 __m512 y = _mm512_div_ps(_mm512_sub_ps(mantissa, _mm512_set1_ps(1.0f)),
684 _mm512_add_ps(mantissa, _mm512_set1_ps(1.0f)));
685
686 __m512 y2 = _mm512_mul_ps(y, y);
687
688 // Polynomial approximation
689 __m512 log_mantissa = _mm512_mul_ps(
690 y,
691 _mm512_add_ps(
692 _mm512_set1_ps(2.0f),
693 _mm512_mul_ps(
694 y2,
695 _mm512_add_ps(
696 _mm512_set1_ps(2.0f / 3.0f),
697 _mm512_mul_ps(
698 y2, _mm512_add_ps(
699 _mm512_set1_ps(2.0f / 5.0f),
700 _mm512_mul_ps(
701 y2, _mm512_set1_ps(2.0f / 7.0f))))))));
702
703 // Convert exponent to float and multiply by log(2)
704 __m512 exponent_f = _mm512_cvtepi32_ps(exponent_adj);
705 __m512 result = _mm512_add_ps(
706 log_mantissa,
707 _mm512_mul_ps(exponent_f, _mm512_set1_ps(0.6931471805599453f)));
708
709 // Handle non-positive inputs
710 result = _mm512_mask_blend_ps(
711 mask_nonpos, result,
712 _mm512_set1_ps(-std::numeric_limits<float>::infinity()));
713
714 _mm512_store_ps(data + i, result);
715 }
716
717 // Process remaining elements
718 for (; i < count; i++) {
719 data[i] = fast_log(data[i]);
720 }
721#elif defined(BREZEL_SIMD_AVX2)
722 // Process 8 elements at a time with AVX2
723 size_t vec_count = count / 8;
724 size_t i = 0;
725
726 for (; i < vec_count * 8; i += 8) {
727 __m256 x = _mm256_load_ps(data + i);
728
729 // Handle non-positive values
730 __m256 zero = _mm256_setzero_ps();
731 __m256 mask_nonpos = _mm256_cmp_ps(x, zero, _CMP_LE_OS);
732
733 // Extract mantissa and exponent using bit manipulation
734 __m256i xi = _mm256_castps_si256(x);
735 __m256i exponent = _mm256_srli_epi32(
736 _mm256_and_si256(xi, _mm256_set1_epi32(0x7F800000)), 23);
737 __m256i mantissa_bits =
738 _mm256_and_si256(xi, _mm256_set1_epi32(0x007FFFFF));
739 __m256 mantissa = _mm256_or_ps(_mm256_castsi256_ps(mantissa_bits),
740 _mm256_set1_ps(1.0f));
741
742 // Scale mantissa to [1, 2) range
743 __m256 one = _mm256_set1_ps(1.0f);
744 __m256 scale_mask = _mm256_cmp_ps(mantissa, one, _CMP_LT_OS);
745 __m256 scaled_mantissa = _mm256_mul_ps(mantissa, _mm256_set1_ps(2.0f));
746 mantissa = _mm256_blendv_ps(mantissa, scaled_mantissa, scale_mask);
747
748 __m256i one_i = _mm256_set1_epi32(1);
749 __m256i adj_exponent = _mm256_sub_epi32(exponent, one_i);
750 exponent = _mm256_blendv_epi8(exponent, adj_exponent,
751 _mm256_castps_si256(scale_mask));
752
753 // y = (mantissa - 1) / (mantissa + 1)
754 __m256 num = _mm256_sub_ps(mantissa, one);
755 __m256 denom = _mm256_add_ps(mantissa, one);
756 __m256 y = _mm256_div_ps(num, denom);
757
758 __m256 y2 = _mm256_mul_ps(y, y);
759
760 // Polynomial approximation
761 __m256 poly_term1 = _mm256_set1_ps(2.0f / 3.0f);
762 __m256 poly_term2 = _mm256_set1_ps(2.0f / 5.0f);
763 __m256 poly_term3 = _mm256_set1_ps(2.0f / 7.0f);
764
765 __m256 poly = _mm256_add_ps(
766 _mm256_set1_ps(2.0f),
767 _mm256_mul_ps(
768 y2,
769 _mm256_add_ps(
770 poly_term1,
771 _mm256_mul_ps(
772 y2, _mm256_add_ps(poly_term2,
773 _mm256_mul_ps(y2, poly_term3))))));
774
775 __m256 log_mantissa = _mm256_mul_ps(y, poly);
776
777 // Convert exponent to float and multiply by log(2)
778 __m256 exponent_f = _mm256_cvtepi32_ps(exponent);
779 __m256 result = _mm256_add_ps(
780 log_mantissa,
781 _mm256_mul_ps(exponent_f, _mm256_set1_ps(0.6931471805599453f)));
782
783 // Handle non-positive inputs
784 __m256 neg_inf =
785 _mm256_set1_ps(-std::numeric_limits<float>::infinity());
786 result = _mm256_blendv_ps(result, neg_inf, mask_nonpos);
787
788 _mm256_store_ps(data + i, result);
789 }
790
791 // Process remaining elements
792 for (; i < count; i++) {
793 data[i] = fast_log(data[i]);
794 }
795#endif
796}
