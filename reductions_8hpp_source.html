154class BREZEL_API MaxFunction : public GradFunctionImpl<MaxFunction> {
155public:
156 MaxFunction(int64_t dim = -1) : m_dim(dim) {
157 m_op_type = OpType::Max;
158 m_name = "MaxFunction";
159 }
160
161 tensor::Tensor<float> forward(
162 const std::vector<tensor::Tensor<float>>& inputs,
163 AutogradContext& ctx) override {
164 BREZEL_ENSURE(inputs.size() == 1, "MaxFunction expects 1 input, got {}",
165 inputs.size());
166
167 auto input = inputs[0];
168
169 ctx.save_for_backward("input", input);
170 ctx.save_for_backward("dim", m_dim);
171
172 return input.max(m_dim);
173 }
174
175 std::vector<tensor::Tensor<float>> backward_impl(
176 const tensor::Tensor<float>& grad_output,
177 const AutogradContext& ctx) override {
178 const auto& input = ctx.get_saved_tensor("input");
179 const auto dim = ctx.get_saved_scalar("dim");
180
181 if (dim == -1) {
182 auto max_val = input.max().data()[0];
183 auto ones = tensor::Tensor<float>::ones(input.shape());
184 auto max_tensor = tensor::Tensor<float>({1}, max_val);
185 auto equals_max = input.subtract(max_tensor).abs().clip(0.0f, 1e-5);
186 auto mask = ones.subtract(equals_max.multiply(
187 tensor::Tensor<float>({1}, 1e5)))
188 .clip(0.0f, 1.0f);
189
190 auto grad_scalar =
191 tensor::Tensor<float>({1}, grad_output.data()[0]);
192 auto grad_input = mask.multiply(grad_scalar);
193
194 return {grad_input};
195 } else {
196 auto max_vals = input.max(dim);
197 auto grad_input = tensor::Tensor<float>::zeros(input.shape());
198 const size_t n = input.numel();
199 const size_t ndim = input.ndim();
200
201 for (size_t i = 0; i < n; ++i) {
202 std::vector<int64_t> indices(ndim);
203 size_t temp = i;
204
205 for (size_t j = 0; j < ndim; ++j) {
206 indices[j] = temp % input.shape()[j];
207 temp /= input.shape()[j];
208 }
209
210 std::vector<int64_t> max_indices = indices;
211 max_indices.erase(max_indices.begin() + dim);
212
213 if (std::abs(input.at(indices) - max_vals.at(max_indices)) <
214 1e-5) {
215 // Max, propagate gradient
216 std::vector<int64_t> grad_indices = max_indices;
217 grad_input.at(indices) = grad_output.at(grad_indices);
218 }
219 }
220
221 return {grad_input};
222 }
223 }
224
225private:
226 int64_t m_dim;
