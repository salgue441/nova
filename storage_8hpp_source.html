588 void resize(size_t new_size) override {
589 if (new_size == m_size) {
590 return;
591 }
592
593 if (has_flag(m_flags, StorageFlags::ReadOnly)) {
594 throw core::error::LogicError("Cannot resize read-only storage");
595 }
596
597 if (m_size == 0) {
598 *this = Storage(new_size, m_flags);
599 return;
600 }
601
602 if (new_size == 0) {
603 if (m_owns_data) {
604 destroy_and_deallocate();
605 }
606
607 m_data = nullptr;
608 m_size = 0;
609 return;
610 }
611
612 pointer new_data = nullptr;
613 if (has_flag(m_flags, StorageFlags::Pooled)) {
614 new_data = MemoryPool::instance().allocate_typed<T>(new_size);
615 } else {
616 AllocatorFactory::AllocatorType alloc_type =
617 has_flag(m_flags, StorageFlags::SimdAligned)
618 ? AllocatorFactory::AllocatorType::Simd
619 : AllocatorFactory::AllocatorType::CacheAligned;
620
621 Allocator<T> allocator(alloc_type);
622 new_data = allocator.allocate(new_size);
623 }
624
625 const size_t copy_size = std::min(m_size, new_size);
626 if (copy_size > 0) {
627 std::uninitialized_copy_n(m_data, copy_size, new_data);
628 }
629
630 if (new_size > m_size &&
631 !has_flag(m_flags, StorageFlags::Uninitialized)) {
632 std::uninitialized_default_construct_n(new_data + m_size,
633 new_size - m_size);
634 }
635 if (m_owns_data) {
636 destroy_and_deallocate();
637 }
638
639 m_data = new_data;
640 m_size = new_size;
641 m_owns_data = true;
642 }
