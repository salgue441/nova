183 BREZEL_NODISCARD Storage(
184 pointer data, size_t size, bool copy_data = true,
185 memory::IAllocator& allocator =
186 memory::MemoryManager::instance().default_allocator(),
187 DeviceType device = DeviceType::CPU)
188 : m_size(size), m_device(device), m_allocator(&allocator) {
189 if (data == nullptr || size == 0) {
190 m_size = 0;
191 return;
192 }
193
194 if (copy_data) {
195 size_t byte_size = size * sizeof(T);
196 m_data = memory::AllocatedPtr<T>(
197 static_cast<T*>(
198 allocator.allocate(byte_size, memory::Alignment::L1Cache)),
199 &allocator, byte_size);
200
201 if (!m_data) {
202 throw core::error::RuntimeError(
203 "Failed to allocate storage of size {}", size);
204 }
205
206 std::memcpy(m_data.get(), data, byte_size);
207 } else {
208 // Take ownership without copying
209 // NOTE: This is dangerous as we don't know how the memory was
210 // allocated This should only be used when the memory was allocated
211 // using the same allocator
212 m_data =
213 memory::AllocatedPtr<T>(data, &allocator, size * sizeof(T));
214 }

371 void resize(size_t new_size, bool preserve_data = true) override {
372 if (new_size == m_size) {
373 return;
374 }
375
376 if (new_size == 0) {
377 m_data.reset();
378 m_size = 0;
379
380 return;
381 }
382
383 if (m_size == 0 || !m_data) {
384 *this = Storage<T>(new_size, *m_allocator, m_device);
385 return;
386 }
387
388 // Check if allocator supports resize
389 size_t old_bytes = m_size * sizeof(T);
390 size_t new_bytes = new_size * sizeof(T);
391
392 if (m_allocator->supports_resize()) {
393 void* new_memory = m_allocator->resize(
394 m_data.get(), old_bytes, new_bytes,
395 preserve_data ? memory::AllocFlags::None
396 : memory::AllocFlags::ZeroMemory);
397
398 if (new_memory) {
399 m_data.reset(static_cast<T*>(new_memory), m_allocator,
400 new_bytes);
401 m_size = new_size;
402
403 return;
404 }
405 }
406
407 memory::AllocatedPtr<T> new_data(
408 static_cast<T*>(m_allocator->allocate(
409 new_bytes, memory::Alignment::L1Cache,
410 preserve_data ? memory::AllocFlags::None
411 : memory::AllocFlags::ZeroMemory)),
412 m_allocator, new_bytes);
413
414 if (!new_data) {
415 throw core::error::RuntimeError("Failed to resize storage to {}",
416 new_size);
417 }
418
419 if (preserve_data) {
420 std::memcpy(new_data.get(), m_data.get(),
421 std::min(old_bytes, new_bytes));
422 }
423
424 m_data = std::move(new_data);
425 m_size = new_size;
426 }

545 BREZEL_NODISCARD SharedStorage(const SharedStorage& other)
546 : m_storage(other.m_storage), m_ref_count(other.m_ref_count) {

548 m_ref_count->fetch_add(1, std::memory_order_relaxed);

555 BREZEL_NODISCARD SharedStorage(SharedStorage&& other) noexcept
556 : m_storage(other.m_storage), m_ref_count(other.m_ref_count) {
557 other.m_storage = nullptr;
558 other.m_ref_count = nullptr;
559 }
560
564 ~SharedStorage() {
565 if (m_ref_count &&
566 m_ref_count->fetch_sub(1, std::memory_order_acq_rel) == 1) {
567 delete m_storage;
568 delete m_ref_count;
569 }
570 }
571
575 SharedStorage& operator=(const SharedStorage& other) {
576 if (this != &other) {
577 if (other.m_ref_count) {
578 other.m_ref_count->fetch_add(1, std::memory_order_relaxed);
579 }
580
581 if (m_ref_count &&
582 m_ref_count->fetch_sub(1, std::memory_order_acq_rel) == 1) {
583 delete m_storage;
584 delete m_ref_count;
585 }
586
587 m_storage = other.m_storage;
588 m_ref_count = other.m_ref_count;
589 }
590
591 return *this;
592 }
593
597 SharedStorage& operator=(SharedStorage&& other) noexcept {
598 if (this != &other) {
599 if (m_ref_count &&
600 m_ref_count->fetch_sub(1, std::memory_order_acq_rel) == 1) {
601 delete m_storage;
602 delete m_ref_count;
603 }
604
605 m_storage = other.m_storage;
606 m_ref_count = other.m_ref_count;
607 other.m_storage = nullptr;
608 other.m_ref_count = nullptr;
