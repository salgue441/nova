41 BREZEL_NODISCARD explicit Strides(const Shape& shape) {
42 if (shape.empty()) {
43 return;
44 }
45
46 const size_t ndim = shape.size();
47 m_strides.resize(ndim);
48
49 if (ndim == 1) {
50 m_strides[0] = 1;
51 return;
52 }
53
54 m_strides[ndim - 1] = 1;
55 for (int64_t i = static_cast<int64_t>(ndim) - 2; i >= 0; --i) {
56 m_strides[i] = m_strides[i + 1] * shape[i + 1];
57 }
58 }

172 BREZEL_NODISCARD BREZEL_FORCE_INLINE size_t
173 get_linear_index(std::span<const int64_t> indices) const {
174 if (indices.size() != size()) {
175 throw core::error::LogicError("Expected {} indices but got {}",
176 size(), indices.size());
177 }
178
179 if (empty()) {
180 return 0;
181 }
182
183 // Optimize for common tensor dimensions
184 if (size() == 1) {
185 return indices[0] * m_strides[0];
186 } else if (size() == 2) {
187 return indices[0] * m_strides[0] + indices[1] * m_strides[1];
188 } else if (size() == 3) {
189 return indices[0] * m_strides[0] + indices[1] * m_strides[1] +
190 indices[2] * m_strides[2];
191 } else if (size() == 4) {
192 return indices[0] * m_strides[0] + indices[1] * m_strides[1] +
193 indices[2] * m_strides[2] + indices[3] * m_strides[3];
194 } else {
195 size_t offset = 0;
196 const size_t n = size();
197 size_t i = 0;
198
199 for (; i + 3 < n; i += 4) {
200 offset += indices[i] * m_strides[i] +
201 indices[i + 1] * m_strides[i + 1] +
202 indices[i + 2] * m_strides[i + 2] +
203 indices[i + 3] * m_strides[i + 3];
204 }
205
206 for (; i < n; ++i) {
207 offset += indices[i] * m_strides[i];
208 }
209
210 return offset;
211 }
212 }

237 BREZEL_NODISCARD Container get_indices(size_t linear_index,
238 const Shape& shape) const {
239 if (shape.size() != size()) {
240 throw core::error::LogicError(
241 "Shape dimensions {} don't match stride dimensions {}",
242 shape.size(), size());
243 }
244
245 Container indices(size(), 0);
246
247 if (size() == 0) {
248 return indices;
249 }
250
251 if (size() == 1) {
252 indices[0] = linear_index / m_strides[0];
253 return indices;
254 }
255
256 size_t remaining = linear_index;
257 for (size_t i = 0; i < size(); ++i) {
258 if (m_strides[i] > 0) {
259 indices[i] = remaining / m_strides[i];
260 remaining %= m_strides[i];
261 }
262 }
263
264 return indices;

434 BREZEL_NODISCARD static Strides broadcast(const Shape& shape,
435 const Shape& target_shape) {
436 if (!shape.is_broadcastable_with(target_shape)) {
437 throw core::error::LogicError(
438 "Shapes {} and {} cannot be broadcast together",
439 shape.to_string(), target_shape.to_string());
440 }
441
442 Strides result;
443 result.m_strides.resize(target_shape.size(), 0);
444
445 if (shape.empty()) {
446 return result;
447 }
448
449 int64_t shape_idx = shape.size() - 1;
450 int64_t target_idx = target_shape.size() - 1;
451 Strides orig_strides(shape);
452
453 while (shape_idx >= 0 && target_idx >= 0) {
454 if (shape[shape_idx] == target_shape[target_idx]) {
455 result.m_strides[target_idx] = orig_strides[shape_idx];
456 } else if (shape[shape_idx] == 1) {
457 result.m_strides[target_idx] = 0;
458 } else {
459 throw core::error::LogicError(
460 "Cannot broadcast dimension {} of shape {} to {}",
461 shape_idx, shape[shape_idx], target_shape[target_idx]);
462 }
463
464 --shape_idx;
465 --target_idx;
466 }
467
468 while (target_idx >= 0) {
469 result.m_strides[target_idx] = 0;
470 --target_idx;
471 }
472
473 return result;
474 }
