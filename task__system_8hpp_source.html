352 bool wait_for(const std::string& task_id, long timeout_ms = 0) {
353 std::unique_lock<std::mutex> lock(m_mutex);
354
355 auto it = m_tasks.find(task_id);
356 if (it == m_tasks.end()) {
357 throw core::error::InvalidArgument("Unknown task ID '{}'", task_id);
358 }
359
360 auto& task = it->second;
361 if (task->status() == TaskStatus::Completed ||
362 task->status() == TaskStatus::Failed ||
363 task->status() == TaskStatus::Cancelled) {
364 return true;
365 }
366
367 if (timeout_ms > 0) {
368 auto predicate = [this, &task_id]() {
369 auto it = m_tasks.find(task_id);
370 if (it == m_tasks.end()) {
371 return true;

375 it->second->status() == TaskStatus::Failed ||
376 it->second->status() == TaskStatus::Cancelled;
377 };
378
379 return m_condition.wait_for(
380 lock, std::chrono::milliseconds(timeout_ms), predicate);
381 } else {
382 m_condition.wait(lock, [this, &task_id]() {
383 auto it = m_tasks.find(task_id);
384 if (it == m_tasks.end()) {
385 return true;
386 }

389 it->second->status() == TaskStatus::Failed ||
390 it->second->status() == TaskStatus::Cancelled;

542 void execute_task(TaskPtr task) {
543 task->set_status(TaskStatus::Running);
544 ThreadPool::get_instance().submit([this, task]() {
545 try {
546 task->execute();
547
548 std::lock_guard<std::mutex> lock(m_mutex);
549 task->set_status(TaskStatus::Completed);
550
551 process_dependents(task->info().id);
552 m_condition.notify_all();
553 } catch (const std::exception& e) {
554 std::lock_guard<std::mutex> lock(m_mutex);
555
556 task->set_status(TaskStatus::Failed);
557 task->set_error(e.what());
558
559 m_condition.notify_all();
560 if (task->info().propagate_exceptions) {
561 throw;
562 }
563 } catch (...) {
564 std::lock_guard<std::mutex> lock(m_mutex);
565
566 task->set_status(TaskStatus::Failed);
567 task->set_error("Unknown exception");
568
569 m_condition.notify_all();
570 if (task->info().propagate_exceptions) {
571 throw;
572 }
573 }
574 });
575 }

582 void process_dependents(const std::string& task_id) {
583 auto it = m_dependents.find(task_id);
584 if (it == m_dependents.end()) {
585 return;
586 }
587
588 for (const auto& dep_task_id : it->second) {
589 auto task_it = m_tasks.find(dep_task_id);
590 if (task_it == m_tasks.end()) {
591 continue;
592 }
593
594 auto& task = task_it->second;
595 bool can_execute = true;
596
597 for (const auto& dep_id : task->info().dependencies) {
598 auto dep_it = m_tasks.find(dep_id);
599 if (dep_it == m_tasks.end()) {
600 can_execute = false;
601 break;
602 }
603
604 if (dep_it->second->status() != TaskStatus::Completed) {
605 can_execute = false;
606 break;
607 }
608 }
609
610 if (can_execute) {
611 execute_task(task);
612 }
613 }
614
615 m_dependents.erase(task_id);
616 }
