726 BREZEL_NODISCARD Tensor narrow(int64_t dim, int64_t start,
727 int64_t length) const {
728 BREZEL_ENSURE(dim >= 0 && dim < static_cast<int64_t>(ndim()),
729 "Dimension out of bounds");
730
731 BREZEL_ENSURE(start >= 0 && start < m_shape[dim],
732 "Start index out of bounds");
733
734 BREZEL_ENSURE(length > 0 && start + length <= m_shape[dim],
735 "Length out of bounds");
736
737 Shape new_shape = m_shape;
738 new_shape[dim] = length;
739
740 Tensor result(new_shape);
741
742 tbb::parallel_for(
743 tbb::blocked_range<size_t>(0, result.numel(), kDefaultBlockSize),
744 [&](const tbb::blocked_range<size_t>& range) {
745 for (size_t i = range.begin(); i < range.end(); ++i) {
746 boost::container::small_vector<int64_t, 4> indices(ndim());
747 size_t temp = i;
748
749 for (size_t j = 0; j < ndim(); ++j) {
750 indices[j] = temp % new_shape[j];
751 temp /= new_shape[j];
752 }
753
754 indices[dim] += start;
755 result.data()[i] = at(indices);
756 }
757 });
758
759 return result;

979 slice(std::span<const std::pair<int64_t, int64_t>> indices) const {
980 BREZEL_ENSURE(indices.size() == ndim(),
981 "Expected {} indices but got {}", ndim(), indices.size());
982
983 for (size_t i = 0; i < indices.size(); ++i) {
984 BREZEL_ENSURE(
985 indices[i].first >= 0 && indices[i].second < m_shape[i],
986 "Index {} out of bounds for dimension {}", indices[i], i);
987 }
988
989 Shape new_shape = m_shape;
990 for (size_t i = 0; i < indices.size(); ++i)
991 new_shape[i] = indices[i].second - indices[i].first;
992
993 Tensor result(new_shape);
994 const size_t n = result.numel();
995
996 tbb::parallel_for(
997 tbb::blocked_range<size_t>(0, n, kDefaultBlockSize),
998 [&](const tbb::blocked_range<size_t>& range) {
999 for (size_t i = range.begin(); i < range.end(); ++i) {
1000 boost::container::small_vector<int64_t, 4> new_indices(
1001 ndim());
1002 size_t temp = i;
1003
1004 for (size_t j = 0; j < ndim(); ++j) {
1005 new_indices[j] = temp % new_shape[j];
1006 temp /= new_shape[j];
1007 }
1008

1010 ndim());
1011 for (size_t j = 0; j < ndim(); ++j)
1012 old_indices[j] = new_indices[j] + indices[j].first;
1013
1014 result.data()[i] = at(old_indices);

1324 const size_t n = m_shape[1];

1335 T h = 0;

1443 if (d[i] != 0) {
1444 for (size_t j = i; j < n; j++) {
1445 T f = 0;
1446 for (size_t k = i; k < n; k++) {

1502 T z = u.data()[j * n + i];

1519 size_t max_idx = i;
1520 T max_val = s.data()[i];

1524 max_val = s.data()[j];

1527 }

1533 std::swap(u.data()[j * n + i], u.data()[j * n + max_idx]);

1536 for (size_t j = 0; j < n; j++) {
1537 std::swap(v.data()[j * n + i], v.data()[j * n + max_idx]);
1538 }
1539 }
1540 }
