184template <TensorScalar T>
185TensorView<T> Tensor<T>::reshape(const Shape& new_shape) const {
186 if (new_shape.numel() != numel()) {
187 throw core::error::LogicError(
188 "Cannot reshape tensor of size {} to size {}", numel(),
189 new_shape.numel());
190 }
191
192 if (is_contiguous()) {
193 auto new_layout = m_layout.reshape(new_shape);
194 return TensorView<T>(*this, new_layout);
195 }
196
197 return contiguous().reshape(new_shape);
198}

200template <TensorScalar T>
201TensorView<T> Tensor<T>::transpose(int64_t dim0, int64_t dim1) const {
202 const size_t dims = ndim();
203
204 if (dim0 < 0)
205 dim0 += dims;
206
207 if (dim1 < 0)
208 dim1 += dims;
209
210 if (dim0 < 0 || dim0 >= static_cast<int64_t>(dims) || dim1 < 0 ||
211 dim1 >= static_cast<int64_t>(dims)) {
212 throw core::error::LogicError(
213 "Dimension out of range for transpose. Got dimensions {} and {} "
214 "for tensor with {} dimensions",
215 dim0, dim1, dims);
216 }
217
218 if (dim0 == dim1) {
219 return TensorView<T>(*this, m_layout);
220 }
221
222 auto new_layout = m_layout.transpose(dim0, dim1);
223 return TensorView<T>(*this, new_layout);
224}

226template <TensorScalar T>
227TensorView<T> Tensor<T>::slice(
228 const std::vector<std::pair<int64_t, int64_t>>& indices) const {
229 const auto& current_shape = shape();
230 const size_t dims = ndim();
231
232 if (indices.size() != dims) {
233 throw core::error::LogicError(
234 "Expected {} indices for slice operation, but got {}", dims,
235 indices.size());
236 }
237
238 Shape new_shape;
239 std::vector<int64_t> new_strides;
240 std::vector<int64_t> offsets;
241
242 for (size_t i = 0; i < dims; ++i) {
243 const auto& [start, end] = indices[i];
244 int64_t dim_size = current_shape[i];
245 int64_t real_start = start < 0 ? start + dim_size : start;
246 int64_t real_end = end < 0 ? end + dim_size : end;
247
248 if (real_start < 0 || real_start >= dim_size) {
249 throw core::error::LogicError(
250 "Slice start index {} out of bounds for dimension {} with size "
251 "{}",
252 start, i, dim_size);
253 }
254
255 if (real_end <= real_start || real_end > dim_size) {
256 throw core::error::LogicError(
257 "Slice end index {} out of bounds for dimension {} with size "
258 "{}",
259 end, i, dim_size);
260 }
261
262 new_shape.push_back(real_end - real_start);
263 new_strides.push_back(strides()[i]);
264 offsets.push_back(real_start * strides()[i]);
265 }
266
267 size_t total_offset =
268 std::accumulate(offsets.begin(), offsets.end(), static_cast<size_t>(0));
269
270 LayoutDescriptor new_layout(new_shape, new_strides);
271 new_layout.set_offset(m_layout.offset() + total_offset);
272 new_layout.set_device(m_layout.device());
273 new_layout.set_format(m_layout.format());
274
275 return TensorView<T>(*this, new_layout);
276}

284template <TensorScalar T>
285TensorView<T> Tensor<T>::squeeze(int64_t dim) const {
286 const auto& current_shape = shape();
287 const size_t dims = ndim();
288
289 if (dim >= 0) {
290 if (dim >= static_cast<int64_t>(dims)) {
291 throw core::error::LogicError(
292 "Dimension {} out of range for tensor with {} dimensions", dim,
293 dims);
294 }
295
296 if (current_shape[dim] != 1) {
297 throw core::error::LogicError(
298 "Cannot squeeze dimension {} with size {}", dim,
299 current_shape[dim]);
300 }
301
302 std::vector<int64_t> new_shape_dims;
303 new_shape_dims.reserve(dims - 1);
304
305 for (size_t i = 0; i < dims; ++i) {
306 if (i != static_cast<size_t>(dim)) {
307 new_shape_dims.push_back(current_shape[i]);
308 }
309 }
310
311 return reshape(Shape(new_shape_dims));
312 } else {
313 std::vector<int64_t> new_shape_dims;
314 new_shape_dims.reserve(dims);
315
316 for (size_t i = 0; i < dims; ++i) {
317 if (current_shape[i] != 1) {
318 new_shape_dims.push_back(current_shape[i]);
319 }
320 }
321
322 if (new_shape_dims.empty()) {
323 return reshape(Shape{});
324 }
325
326 return reshape(Shape(new_shape_dims));
327 }
328}

330template <TensorScalar T>
331TensorView<T> Tensor<T>::unsqueeze(int64_t dim) const {
332 const auto& current_shape = shape();
333 const size_t dims = ndim();
334
335 if (dim < 0) {
336 dim += dims + 1;
337 }
338
339 if (dim < 0 || dim > static_cast<int64_t>(dims)) {
340 throw core::error::LogicError(
341 "Dimension {} out of range for unsqueeze on tensor with {} "
342 "dimensions",
343 dim, dims);
344 }
345
346 std::vector<int64_t> new_shape_dims;
347 new_shape_dims.reserve(dims + 1);
348
349 for (size_t i = 0; i < dims + 1; ++i) {
350 if (i == static_cast<size_t>(dim)) {
351 new_shape_dims.push_back(1);
352 }
353
354 if (i < dims) {
355 new_shape_dims.push_back(current_shape[i]);
356 }
357 }
358
359 return reshape(Shape(new_shape_dims));
360}
