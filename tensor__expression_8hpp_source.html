289 std::vector<int64_t> map_indices_for_broadcast(
290 std::span<const int64_t> output_indices,
291 const Shape& input_shape) const {
292 const size_t out_dims = output_indices.size();
293 const size_t in_dims = input_shape.size();
294
295 std::vector<int64_t> input_indices(in_dims);
296 for (size_t i = 0; i < in_dims; ++i) {
297 const size_t out_dim = out_dims - in_dims + i;
298
299 if (out_dim < out_dims) {
300 if (input_shape[i] == 1) {
301 input_indices[i] = 0;
302 } else {
303 input_indices[i] = output_indices[out_dim];
304 }
305 } else {
306 input_indices[i] = 0;
307 }
308 }
309
310 return input_indices;
311 }
