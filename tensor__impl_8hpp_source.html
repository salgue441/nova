226 explicit Tensor(const detail::TensorExpression<T>& expr)
227 : m_layout(expr.shape()),
228 m_storage(std::make_shared<detail::TensorStorage<T>>(expr.numel())) {
229 const size_t size = numel();
230 pointer dst = data();
231 const Shape& result_shape = shape();
232
233 std::vector<int64_t> indices(ndim(), 0);
234 for (size_t i = 0; i < size; ++i) {
235 size_t idx = i;
236
237 for (int64_t d = ndim() - 1; d >= 0; --d) {
238 indices[d] = idx % result_shape[d];
239 idx /= result_shape[d];
240 }
241
242 dst[i] = expr.eval(indices);
243 }
244 }

451 static Tensor linspace(T start, T end, int64_t steps) {
452 if (steps <= 0) {
453 throw core::error::InvalidArgument(
454 "Number of steps must be positive");
455 }
456
457 Tensor result({steps});
458 T* data_ptr = result.data();
459
460 if (steps == 1) {
461 data_ptr[0] = start;
462 return result;
463 }
464
465 const T step = (end - start) / static_cast<T>(steps - 1);
466 tbb::parallel_for(
467 tbb::blocked_range<int64_t>(0, steps),
468 [=](const tbb::blocked_range<int64_t>& range) {
469 for (int64_t i = range.begin(); i < range.end(); ++i) {
470 data_ptr[i] = start + step * static_cast<T>(i);
471 }
472 });
473
474 return result;
475 }

494 static Tensor arange(T start, T end, T step = T(1)) {
495 if (step == T(0)) {
496 throw core::error::InvalidArgument("Step cannot be zero");
497 }
498
499 if ((step > T(0) && start >= end) || (step < T(0) && start <= end)) {
500 return Tensor({0});
501 }
502
503 const int64_t steps =
504 static_cast<int64_t>(std::ceil((end - start) / step));
505 Tensor result({steps});
506 T* data_ptr = result.data();
507
508 tbb::parallel_for(
509 tbb::blocked_range<int64_t>(0, steps),
510 [=](const tbb::blocked_range<int64_t>& range) {
511 for (int64_t i = range.begin(); i < range.end(); ++i) {
512 data_ptr[i] = start + step * static_cast<T>(i);
513 }
514 });

516 return result;

540 static Tensor random(const Shape& shape, T min = T(0), T max = T(1),
541 MemoryLayout layout = MemoryLayout::RowMajor) {
542 Tensor result(shape, layout);
543 T* data_ptr = result.data();
544 const size_t size = result.numel();
545
546 if constexpr (!std::is_arithmetic_v<T>) {
547 static_assert(std::is_arithmetic_v<T>,
548 "random() requires arithmetic type");
549
550 return result;

554 max);
555 tbb::parallel_for(
556 tbb::blocked_range<size_t>(0, size, 1024),
557 [&](const tbb::blocked_range<size_t>& range) {
558 auto gen = generator_factory.get_generator(
559 std::hash<std::thread::id>{}(std::this_thread::get_id()));
560

562 data_ptr[i] = gen();
563 }

701 // Clone implementation (deep copy)
714 Tensor clone() const override {
715 Tensor result(shape(), layout().layout());
716 const size_t size = numel();
717
718 if (is_contiguous() && result.is_contiguous()) {
719 std::memcpy(result.data(), data(), size * sizeof(T));
720 } else {
721 auto iter = m_layout.create_iterator();
722 auto result_iter = result.layout().create_iterator();
723
724 for (size_t i = 0; i < size; ++i) {
725 result.data()[result_iter.offset()] = data()[iter.offset()];
726 iter.next();
727 result_iter.next();
728 }
729 }
730
731 return result;

893 Tensor<T> contiguous() const {
894 if (is_contiguous()) {
895 return *this;
896 }
897
898 Tensor result(shape());
899 const auto size = numel();
900
901 auto iter = m_layout.create_iterator();
902 T* dst = result.data();
903 const T* src = data();
904
905 if (size > 1024) {
906 tbb::parallel_for(tbb::blocked_range<size_t>(0, size),
907 [&](const tbb::blocked_range<size_t>& range) {
908 auto local_iter = m_layout.create_iterator();
909 for (size_t i = 0; i < range.begin(); ++i) {
910 local_iter.next();
911 }
912
913 for (size_t i = range.begin();
914 i < range.end(); ++i) {
915 dst[i] = src[local_iter.offset()];
916 local_iter.next();
917 }
918 });
919 } else {
920 for (size_t i = 0; i < size; ++i) {
921 dst[i] = src[iter.offset()];
922 iter.next();
923 }
924 }
925
926 return result;
927 }
