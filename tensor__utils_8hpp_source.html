77template <typename U, TensorScalar T>
78size_t fill_from_nested_list(const U& data, T* ptr,
79 const std::vector<int64_t>& dims, size_t dim_idx,
80 size_t offset) {
81 if (dim_idx == dims.size() - 1) {
82 size_t i = 0;
83
84 for (const auto& val : data) {
85 if (i >= static_cast<size_t>(dims[dim_idx])) {
86 break;
87 }
88
89 ptr[offset + i] = static_cast<T>(val);
90 ++i;
91 }
92
93 return offset + dims[dim_idx];
94 } else {
95 size_t i = 0;
96 size_t new_offset = offset;
97 size_t stride = 1;
98
99 for (size_t j = dim_idx + 1; j < dims.size(); ++i) {
100 stride *= dims[j];
101 }
102
103 for (const auto& sublist : data) {
104 if (i >= static_cast<size_t>(dims[dim_idx])) {
105 break;
106 }
107
108 new_offset = fill_from_nested_list(sublist, ptr, dims, dim_idx + 1,
109 offset + i * stride);
110 ++i;
111 }
112
113 return new_offset;
114 }
115}

140template <TensorScalar T>
141std::string format_element(const T& value, int precision = 4) {
142 std::ostringstream oss;
143
144 if constexpr (std::is_floating_point_v<T>) {
145 oss << std::fixed << std::setprecision(precision) << value;
146
147 std::string str = oss.str();
148 size_t dot_pos = str.find('.');
149
150 if (dot_pos != std::string::npos) {
151 size_t last_non_zero = str.find_last_not_of('0');
152 if (last_non_zero != std::string::npos && last_non_zero > dot_pos) {
153 str.erase(last_non_zero + 1);
154 } else if (last_non_zero == dot_pos) {
155 str.erase(dot_pos);
156 }
157 }
158
159 return str;
160 } else if constexpr (std::is_integral_v<T>) {
161 oss << value;
162 return oss.str();
163 } else if constexpr (std::is_same_v<T, bool>) {
164 return value ? "true" : "false";
165 } else {
166 oss << value;
167 return oss.str();
168 }
169}

195template <TensorScalar T>
196void print_tensor_recursive(std::ostream& os, const T* data,
197 const LayoutDescriptor& layout,
198 const std::vector<int64_t>& dims, size_t dim_idx,
199 std::vector<int64_t>& current_indices,
200 int max_per_line, int precision) {
201 if (dim_idx == dims.size() - 1) {
202 oss << "[";
203
204 bool truncated = false;
205 size_t line_size = dims[dim_idx];
206
207 if (max_per_line > 0 &&
208 line_size > static_cast<size_t>(2 * max_per_line + 1)) {
209 line_size = 2 * max_per_line + 1;
210 truncated = true;
211 }
212
213 for (size_t i = 0; i < line_size; ++i) {
214 if (truncated && i == max_per_line) {
215 os << " ... ";
216
217 i = dims[dim_idx] - max_per_line - 1;
218 continue;
219 }
220
221 current_indices[dim_idx] = static_cast<int64_t>(i);
222 size_t linear_index = layout.get_linear_index(current_indices);
223
224 os << format_element(data[linear_idx], precision);
225 if (i < line_size - 1) {
226 os << ", ";
227 }
228 }
229
230 os << "]";
231 } else {
232 os << "[";
233
234 bool truncated = false;
235 size_t subarray_count = dims[dim_idx];
236
237 if (max_per_line > 0 &&
238 subarray_count > static_cast<size_t>(2 * max_per_line + 1)) {
239 subarray_count = 2 * max_per_line + 1;
240 truncated = true;
241 }
242
243 for (size_t i = 0; i < subarray_count; ++i) {
244 if (truncated && i == max_per_line) {
245 os << " ... ";
246
247 i = dims[dim_idx] - max_per_line - 1;
248 continue;
249 }
250
251 current_indices[dim_idx] = static_cast<int64_t>(i);
252 print_tensor_recursive(os, data, layout, dims, dim_idx + 1,
253 current_indices, max_per_line, precision);
254
255 if (i < subarray_count - 1) {
256 os << ",\n" << std::string(dim_idx + 1, ' ');
257 }
258 }
259
260 os << "]";
261 }

285template <TensorScalar T>
286std::string tensor_to_string(const T* data, const LayoutDescriptor& layout,
287 int max_per_line = 6, int precision = 4) {
288 std::ostringstream os;
289 const auto& shape = layout.shape();
290 std::vector<int64_t> dims;
291
292 dims.reserve(shape.size());
293 for (size_t i = 0; i < shape.size(); ++i) {
294 dims.push_back(shape[i]);
295 }
296
297 if (dims.empty()) {
298 return format_element(data[0], precision);
299 }
300
301 std::vector<int64_t> current_indices(dims.size(), 0);
302 print_tensor_recursive(os, data, layout, dims, 0, current_indices,
303 max_per_line, precision);
304
305 return os.str();
306}
