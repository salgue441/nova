132 void backward(const tensor::Tensor<float>& grad = tensor::Tensor<float>()) {
133 BREZEL_ENSURE(
134 m_requires_grad,
135 "Calling backward on a variable that doesn't require gradients");
136
137 bool is_grad_empty = (grad.numel() == 0 || grad.data() == nullptr);
138 BREZEL_ENSURE(
139 m_data.numel() == 1 || !is_grad_empty,
140 "Gradient can be implicitly created only for scalar outputs");
141
142 tensor::Tensor<float> grad_output = grad;
143 if (is_grad_empty) {
144 BREZEL_ENSURE(m_data.numel() == 1,
145 "Non-scalar output requires explicit gradient");
146
147 grad_output = tensor::Tensor<float>({1}, 1.0f);
148 } else {
149 grad_output = grad;
150 }
151
152 BREZEL_ENSURE(grad_output.shape() == m_data.shape(),
153 "Gradient shape mismatch: expected {}, got {}",
154 m_data.shape().to_string(),
155 grad_output.shape().to_string());
156
157 std::vector<std::shared_ptr<Variable>> execution_plan =
158 build_execution_plan();
159
160 std::unordered_map<std::shared_ptr<Variable>, tensor::Tensor<float>>
161 grads;
162
163 grads[shared_from_this()] = grad_output;
164
165 for (auto& var : execution_plan) {
166 if (!var->requires_grad() || !var->grad_fn())
167 continue;
168
169 auto it = grads.find(var);
170 if (it == grads.end())
171 continue;
172
173 auto& grad_output = it->second;
174 auto grad_fn = var->grad_fn();
175 auto input_grads = grad_fn->backward(grad_output);
176 auto inputs = get_inputs_from_grad_fn(grad_fn);
177
178 for (size_t i = 0; i < inputs.size(); ++i) {
179 if (!inputs[i]->requires_grad())
180 continue;
181
182 auto input_grad = input_grads[i];
183 if (grads.find(inputs[i]) == grads.end())
184 grads[inputs[i]] = input_grad;
185
186 else
187 grads[inputs[i]] = grads[inputs[i]] + input_grad;
188 }
189 }
190
191 for (auto& [var, grad] : grads) {
192 if (var->requires_grad())
193 var->m_grad = grad;
194 }
195 }

226 std::vector<std::shared_ptr<Variable>> build_execution_plan() {
227 std::vector<std::shared_ptr<Variable>> result;
228 std::unordered_set<std::shared_ptr<Variable>> visited;
229 std::function<void(std::shared_ptr<Variable>)> dfs;
230
231 dfs = [&](std::shared_ptr<Variable> var) {
232 if (visited.find(var) != visited.end())
233 return;
234
235 visited.insert(var);
236 auto inputs = get_inputs_from_grad_fn(var->grad_fn());
237
238 for (const auto& input : inputs) {
239 if (input->requires_grad())
240 dfs(input);
241 }
242
243 result.push_back(var);
244 };
245
246 dfs(shared_from_this());
247 return result;
248 }
