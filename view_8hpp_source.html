147 // Clone (materializes the view into a new tensor)
157 Tensor<T> clone() const override {
158 Tensor<T> result(shape(), layout().layout());
159 const size_t size = numel();
160
161 auto iter = m_layout.create_iterator();
162 auto result_iter = result.layout().create_iterator();
163 const_pointer src = m_base_tensor.data();
164 pointer dst = result.data();
165
166 for (size_t i = 0; i < size; ++i) {
167 dst[result_iter.offset()] = src[iter.offset()];
168 iter.next();
169 result_iter.next();
170 }
171
172 return result;
173 }

389 TensorView slice(
390 const std::vector<std::pair<int64_t, int64_t>>& indices) const {
391 const auto& current_shape = shape();
392 const size_t dims = ndim();
393
394 if (indices.size() != dims) {
395 throw core::error::LogicError(
396 "Expected {} indices for slice operation, but got {}", dims,
397 indices.size());
398 }
399
400 Shape new_shape;
401 std::vector<int64_t> new_strides;
402 std::vector<int64_t> offsets;
403
404 for (size_t i = 0; i < dims; ++i) {
405 const auto& [start, end] = indices[i];
406 int64_t dim_size = current_shape[i];
407 int64_t real_start = start < 0 ? start + dim_size : start;
408 int64_t real_end = end < 0 ? end + dim_size : end;
409
410 if (real_start < 0 || real_start >= dim_size) {
411 throw core::error::LogicError(
412 "Slice start index {} out of bounds for dimension {} with "
413 "size {}",
414 start, i, dim_size);
415 }
416
417 if (real_end <= real_start || real_end > dim_size) {
418 throw core::error::LogicError(
419 "Slice end index {} out of bounds for dimension {} with "
420 "size {}",
421 end, i, dim_size);
422 }
423
424 new_shape.push_back(real_end - real_start);
425 new_strides.push_back(m_layout.strides()[i]);
426 offsets.push_back(real_start * m_layout.strides()[i]);
427 }
428
429 size_t total_offset = std::accumulate(offsets.begin(), offsets.end(),
430 static_cast<size_t>(0));
431
432 LayoutDescriptor new_layout(new_shape, new_strides);
433 new_layout.set_offset(m_layout.offset() + total_offset);
434 new_layout.set_device(m_layout.device());
435 new_layout.set_format(m_layout.format());
436
437 return TensorView(m_base_tensor, new_layout);
438 }

470 TensorView squeeze(int64_t dim = -1) const {
471 const auto& current_shape = shape();
472 const size_t dims = ndim();
473
474 if (dim >= 0) {
475 if (dim >= static_cast<int64_t>(dims)) {
476 throw core::error::LogicError(
477 "Dimension {} out of range for tensor with {} dimensions",
478 dim, dims);
479 }
480
481 if (current_shape[dim] != 1) {
482 throw core::error::LogicError(
483 "Cannot squeeze dimension {} with size {}", dim,
484 current_shape[dim]);
485 }
486
487 std::vector<int64_t> new_shape_dims;
488 new_shape_dims.reserve(dims - 1);
489
490 for (size_t i = 0; i < dims; ++i) {
491 if (i != static_cast<size_t>(dim)) {
492 new_shape_dims.push_back(current_shape[i]);
493 }
494 }
495
496 return reshape(Shape(new_shape_dims));
497 } else {
498 std::vector<int64_t> new_shape_dims;
499 new_shape_dims.reserve(dims);
500
501 for (size_t i = 0; i < dims; ++i) {
502 if (current_shape[i] != 1) {
503 new_shape_dims.push_back(current_shape[i]);
504 }
505 }
506
507 if (new_shape_dims.empty()) {
508 return reshape(Shape{});
509 }
510
511 return reshape(Shape(new_shape_dims));
512 }
513 }

531 TensorView unsqueeze(int64_t dim) const {
532 const auto& current_shape = shape();
533 const size_t dims = ndim();
534
535 if (dim < 0) {
536 dim += dims + 1;
537 }
538
539 if (dim < 0 || dim > static_cast<int64_t>(dims)) {
540 throw core::error::LogicError(
541 "Dimension {} out of range for unsqueeze on tensor with {} "
542 "dimensions",
543 dim, dims);
544 }
545
546 std::vector<int64_t> new_shape_dims;
547 new_shape_dims.reserve(dims + 1);
548
549 for (size_t i = 0; i < dims + 1; ++i) {
550 if (i == static_cast<size_t>(dim)) {
551 new_shape_dims.push_back(1);
552 }
553
554 if (i < dims) {
555 new_shape_dims.push_back(current_shape[i]);
556 }
557 }
558
559 return reshape(Shape(new_shape_dims));
560 }

911 BREZEL_NODISCARD inline size_t flat_index_to_offset(size_t idx) const {
912 const Shape& shape_val = shape();
913 std::vector<int64_t> indices(ndim());
914
915 for (int dim = ndim() - 1; dim >= 0; --dim) {
916 indices[dim] = idx % shape_val[dim];
917 idx /= shape_val[dim];
918 }
919
920 return m_layout.get_linear_index(indices);
